    direct_dependents[unit] <-
      COUNT(units WHERE unit IN their_deps)
    transitive_dependents[unit] <-

P1.2  G.6  student_state_model
// bootstrapped on empty student_record
// updates incrementally per evidence

OPERATIONS:

  GET_STATE(unit_id) ->
    student_record.units[unit_id]

  UPDATE_STATE(unit_id, evidence) ->
    state <- GET_STATE(unit_id)
    state.evidence_log.append(evidence)
    state.last_seen <- now()
    state.mastery_level   <- COMPUTE_MASTERY(unit_id)
    state.fluency_level   <- COMPUTE_FLUENCY(unit_id)
    state.retention_score <- COMPUTE_RETENTION(unit_id)

  COMPUTE_MASTERY(unit_id) ->
    log <- evidence_log[unit_id]
    IF log.length < MIN_EVIDENCE: RETURN null
    correct  <- COUNT(log WHERE outcome == correct)
    total    <- log.length
    variety  <- COUNT_DISTINCT(log.format)
    levels   <- COUNT_DISTINCT(log.demand_level)
    RETURN f(correct/total, variety, levels)
    // f weights consistency + variety + level_coverage

  COMPUTE_FLUENCY(unit_id) ->
    log <- evidence_log[unit_id]
      .filter(timed == TRUE)
    IF log.length < MIN_TIMED_EVIDENCE: RETURN null
    on_time <- COUNT(log WHERE time <= threshold)
    RETURN on_time / log.length

  COMPUTE_RETENTION(unit_id) ->
    log <- evidence_log[unit_id]
    recent <- log.filter(
      last_seen > now() - RETENTION_WINDOW
    )
    IF recent.empty: RETURN decay(last_score, elapsed)
    RETURN COMPUTE_MASTERY(unit_id)
    // decay: exponential, calibrated by D.6

  BASE_STABLE(unit_id) ->
    deps <- GET_DEPS(unit_id)  // from P0.2
    FOR each d IN deps:
      IF GET_STATE(d).mastery_level < MASTERY_THRESHOLD:
        RETURN FALSE
    RETURN TRUE

CONSTANTS:
  MIN_EVIDENCE:       5
  MIN_TIMED_EVIDENCE: 3
  MASTERY_THRESHOLD:  0.80
  FLUENCY_THRESHOLD:  0.75
  RETENTION_WINDOW:   14 days
  RETENTION_DECAY:    exponential(halflife=21 days)

// gera itens por unidade e nivel de exigencia
// com variacao controlada de formato e contexto

DEMAND_LEVELS:
  L1  recognition       // identificar, distinguir
  L2  guided_execution  // executar com suporte
  L3  free_execution    // executar sem suporte
  L4  transfer          // aplicar em contexto novo
  L5  integration       // mobilizar multiplos conteudos

FORMAT_VARIANTS:
  numeric_direct        // ex: calcule 3/4 + 1/6
  numeric_indirect      // ex: qual fracao equivale a 0.75?
  verbal_simple         // ex: "Pedro tem 3/4 de pizza..."
  symbolic              // ex: simplificar expressao algebrica
  comparison            // ex: ordene: 2/3, 3/4, 5/8
  error_detection       // ex: encontre o erro no calculo
  estimation            // ex: estime sem calcular

CONTEXT_VARIANTS:
  bare                  // sem contexto
  numeric_context       // contexto puramente numerico
  verbal_context        // situacao descrita em texto
  mixed                 // multiplos tipos combinados

GENERATE(unit_id, level, format, context) ->
  template <- SELECT_TEMPLATE(unit_id, level)
  surface  <- APPLY_SURFACE(template, format, context)
  item <- {
    unit_id:      unit_id,
    demand_level: level,
    format:       format,
    context:      context,
    content:      surface,
    timed:        level >= L3 AND unit_in_fluency_scope(unit_id),
    time_limit:   GET_TIME_LIMIT(unit_id, level)
  }
  RETURN item

SELECT_NEXT_ITEM(unit_id, student_state) ->
  level   <- DETERMINE_LEVEL(unit_id, student_state)
  format  <- SELECT_FORMAT(unit_id, student_state)
  context <- SELECT_CONTEXT(unit_id, student_state)
  RETURN GENERATE(unit_id, level, format, context)

DETERMINE_LEVEL(unit_id, state) ->
  m <- state.mastery_level
  IF m == null:                     RETURN L1
  IF m < 0.5:                       RETURN L1
  IF m < 0.70:                      RETURN L2
  IF m < MASTERY_THRESHOLD:         RETURN L3
  IF NOT state.transfer_demonstrated: RETURN L4
  IF NOT state.integration_demonstrated: RETURN L5
  RETURN L3  // maintenance

SELECT_FORMAT(unit_id, state) ->
  used <- COUNT_DISTINCT(state.evidence_log.format)
  IF used < 3: RETURN LEAST_USED_FORMAT(state)
  RETURN RANDOM_FROM(FORMAT_VARIANTS)

SELECT_CONTEXT(unit_id, state) ->
  used <- COUNT_DISTINCT(state.evidence_log.context)
  IF used < 2: RETURN LEAST_USED_CONTEXT(state)
  RETURN RANDOM_FROM(CONTEXT_VARIANTS)

    KC.1-KC.8,     // propriedades basicas
    KD.1-KD.9,     // divisibilidade
    KE.10-KE.16    // conversoes

RECORD_EVIDENCE(unit_id, item, response) ->

    timestamp:    now(),
    demand_level: item.demand_level,
    format:       item.format,
    context:      item.context,
    outcome:      EVALUATE(item, response),
    timed:        item.timed,

MASTERY_CONCLUDED(unit_id) ->
  log   <- evidence_log[unit_id]
  IF log.length < MIN_COUNT:             RETURN FALSE
  IF CONSISTENCY(log) < MIN_CONSISTENCY: RETURN FALSE
  IF COUNT_DISTINCT(log.format) < MIN_VARIETY: RETURN FALSE
  IF COUNT_DISTINCT(log.demand_level) < MIN_LEVELS: RETURN FALSE
  RETURN TRUE

CONSISTENCY(log) ->
  recent <- log.last(MIN_COUNT)
  RETURN COUNT(recent WHERE outcome == correct) / MIN_COUNT

P2.3  G.3  error_classifier
// classifica cada erro em 5 dimensoes

CLASSIFY(unit_id, evidence_log) ->
  IF evidence_log.empty: RETURN null
  errors <- evidence_log.filter(outcome == incorrect)
  IF errors.empty: RETURN null

  RETURN {
    frequency:   CLASSIFY_FREQUENCY(errors, evidence_log),
    scope:       CLASSIFY_SCOPE(unit_id, errors),
    type:        CLASSIFY_TYPE(unit_id, errors),
    transfer:    CLASSIFY_TRANSFER(unit_id, errors),
    integration: CLASSIFY_INTEGRATION(unit_id, errors)
  }

CLASSIFY_FREQUENCY(errors, log) ->
  rate <- errors.length / log.length
  IF rate > 0.4 AND errors.length >= 3:
    RETURN recurrent
  RETURN isolated

CLASSIFY_SCOPE(unit_id, errors) ->
  deps <- GET_DEPS(unit_id)
  failing_deps <- deps.filter(
    GET_STATE(d).mastery_level < 0.5
    OR GET_STATE(d).mastery_level == null
  )
  IF failing_deps.length > 0: RETURN structural
  RETURN local

CLASSIFY_TYPE(unit_id, errors) ->
  conceptual_errors <- errors.filter(
    demand_level IN {L1, L2}
  )
  fluency_errors <- errors.filter(
    timed == TRUE
    AND outcome == timeout OR latency > threshold
  )
  IF conceptual_errors.length > fluency_errors.length:
    RETURN conceptual
  RETURN fluency

CLASSIFY_TRANSFER(unit_id, errors) ->
  transfer_errors <- errors.filter(demand_level == L4)
  trained_errors  <- errors.filter(demand_level IN {L2,L3})
  IF transfer_errors.length > 0
     AND trained_errors.length == 0:
    RETURN transfer_failure
  RETURN trained_format_failure

CLASSIFY_INTEGRATION(unit_id, errors) ->
  integration_errors <- errors.filter(demand_level == L5)
  unit_errors <- errors.filter(demand_level IN {L2,L3})
  IF integration_errors.length > 0
     AND unit_errors.length == 0:
    RETURN integration_failure
  RETURN unit_failure

P2.4  G.4  cause_mapper
// mapeia classificacao -> causa provavel
// usa estrutura D.5

CAUSE_MAP = {
  {recurrent, structural, conceptual, *, *}:
    MISSING_PREREQUISITE,

  {recurrent, local, conceptual, *, *}:
    CONCEPTUAL_GAP,

  {recurrent, local, fluency, *, *}:
    LOW_AUTOMATIZATION,

  {isolated, local, conceptual, *, *}:
    ATTENTION_ERROR,

  {*, *, *, transfer_failure, *}:
    TRANSFER_NOT_DEVELOPED,

  {*, *, *, *, integration_failure}:
    INTEGRATION_NOT_DEVELOPED,

  {recurrent, structural, fluency, *, *}:
    BASE_FLUENCY_INSUFFICIENT
}

MAP_CAUSE(classification) ->
  FOR each pattern IN CAUSE_MAP:
    IF MATCHES(classification, pattern):
      RETURN CAUSE_MAP[pattern]
  RETURN UNKNOWN_CAUSE

MATCHES(classification, pattern) ->
  FOR each dimension IN pattern:
    IF pattern[dimension] != *
       AND classification[dimension] != pattern[dimension]:
      RETURN FALSE
  RETURN TRUE

// bloqueia avanco enquanto base instavel

CAN_ADVANCE(unit_id) ->
  state <- GET_STATE(unit_id)

  // check 1: base deps satisfied
  IF NOT BASE_STABLE(unit_id):
    RETURN BLOCKED(reason=UNSTABLE_BASE,
      failing_deps=GET_FAILING_DEPS(unit_id))

  // check 2: mastery evidence sufficient
  IF NOT MASTERY_CONCLUDED(unit_id):
    RETURN BLOCKED(reason=INSUFFICIENT_EVIDENCE,
      current=state.mastery_level,
      required=MASTERY_THRESHOLD)

  // check 3: retention if previously concluded
  IF state.retention_score != null
     AND state.retention_score < RETENTION_THRESHOLD:
    RETURN BLOCKED(reason=RETENTION_DECAYED,
      current=state.retention_score,
      required=RETENTION_THRESHOLD)

  RETURN OPEN

GET_FAILING_DEPS(unit_id) ->
  deps <- GET_DEPS(unit_id)
  RETURN deps.filter(
    GET_STATE(d).mastery_level == null
    OR GET_STATE(d).mastery_level < MASTERY_THRESHOLD
  )

  RETENTION_THRESHOLD: 0.70

P3.2  G.7  next_content_selector
// seleciona proximo conteudo por estado atual
// balanceia novidade | consolidacao | revisao | retomada

PRIORITIES:
  P_RETAKE:        1  // base instavel: maior prioridade
  P_REVIEW:        2  // retencao em queda
  P_CONSOLIDATE:   3  // mastery < threshold mas iniciado
  P_NEW:           4  // pronto para avancar

SELECT_NEXT(student_record) ->
  candidates <- RANK_CANDIDATES(student_record)
  RETURN candidates.first()

RANK_CANDIDATES(student_record) ->
  scored <- []
  FOR each unit IN unit_list:
    state  <- GET_STATE(unit)
    gate   <- CAN_ADVANCE(unit)
    sv     <- structural_value[unit]  // from G.21
    score  <- COMPUTE_SCORE(unit, state, gate, sv)
    scored.append({unit, score})
  RETURN scored.sort_by(score DESC)

COMPUTE_SCORE(unit, state, gate, sv) ->

  // retake: dep failing
  IF gate.reason == UNSTABLE_BASE:
    failing <- gate.failing_deps
    RETURN P_RETAKE * sv
           + PRIORITY_BOOST * failing.length

  // review: retention decaying
  IF state.retention_score != null
     AND state.retention_score < RETENTION_THRESHOLD:
    decay <- RETENTION_THRESHOLD - state.retention_score
    RETURN P_REVIEW * sv + decay * 10

  // consolidate: started but not concluded
  IF state.mastery_level != null
     AND state.mastery_level < MASTERY_THRESHOLD
     AND gate == OPEN:
    progress <- state.mastery_level
    RETURN P_CONSOLIDATE * sv + progress

  // new: gate open, not yet started
  IF state.mastery_level == null
     AND gate == OPEN:
    RETURN P_NEW * sv

  // already concluded: maintenance only
  IF MASTERY_CONCLUDED(unit)
     AND state.retention_score >= RETENTION_THRESHOLD:
    RETURN 0

  RETURN 0

SESSION_PLAN(student_record, session_size=10) ->
  plan        <- []
  new_count   <- 0
  NEW_LIMIT   <- 2  // E.16: max new content per session

  candidates  <- RANK_CANDIDATES(student_record)

  FOR each candidate IN candidates:
    IF plan.length >= session_size: BREAK

    type <- CLASSIFY_CANDIDATE_TYPE(candidate, student_record)

    IF type == NEW AND new_count >= NEW_LIMIT:
      CONTINUE  // E.16 enforced

    // E.17: cognitive load check
    projected_load <- ESTIMATE_LOAD(plan + candidate)
    IF projected_load > LOAD_THRESHOLD:
      CONTINUE

    plan.append({unit: candidate.unit, type: type})
    IF type == NEW: new_count += 1

  RETURN plan

CLASSIFY_CANDIDATE_TYPE(candidate, student_record) ->
  state <- GET_STATE(candidate.unit)
  IF state.mastery_level == null:               RETURN new
  IF state.retention_score < RETENTION_THRESHOLD: RETURN review
  IF state.mastery_level < MASTERY_THRESHOLD:   RETURN consolidate
  RETURN maintenance

ESTIMATE_LOAD(plan) ->
  total <- 0
  FOR each item IN plan:
    state   <- GET_STATE(item.unit)
    level   <- DETERMINE_LEVEL(item.unit, state)
    novelty <- IF state.mastery_level == null: 1.5 ELSE 1.0
    total   += level * novelty
  RETURN total

CONSTANTS:
  LOAD_THRESHOLD:  25
  PRIORITY_BOOST:  5

INTERVAL_SCHEDULE (from D.6):

  GET_INTERVAL(unit_id) ->
    robustness <- GET_ROBUSTNESS(unit_id)
    base       <- BASE_INTERVALS[robustness_tier(robustness)]
    sv_factor  <- 1 / (1 + structural_value_normalized[unit_id])
    RETURN base * sv_factor

  BASE_INTERVALS = {
    tier_1_robustness:  1 day,
    tier_2_robustness:  3 days,
    tier_3_robustness:  7 days,
    tier_4_robustness:  14 days,
    tier_5_robustness:  21 days
  }

  GET_ROBUSTNESS(unit_id) ->
    log <- evidence_log[unit_id]
      .filter(type == review)
    IF log.length < 3: RETURN tier_1_robustness
    above <- log.filter(
      retention_score >= RETENTION_THRESHOLD
    )
    RETURN TIER_FROM_RATIO(above.length / log.length)

  TIER_FROM_RATIO(ratio) ->
    IF ratio < 0.40: RETURN tier_1_robustness
    IF ratio < 0.60: RETURN tier_2_robustness
    IF ratio < 0.75: RETURN tier_3_robustness
    IF ratio < 0.90: RETURN tier_4_robustness
    RETURN tier_5_robustness

SCHEDULE_REVISION(unit_id) ->
  interval    <- GET_INTERVAL(unit_id)
  next_review <- now() + interval
  student_record.units[unit_id]
    .next_review <- next_review
  RETURN next_review

REVISIONS_DUE(student_record) ->
  RETURN unit_list.filter(
    student_record.units[u].next_review != null
    AND student_record.units[u].next_review <= now()
  ).sort_by(
    student_record.units[u].next_review ASC
  )

PROJECT_RETENTION(unit_id, at_time) ->
  last_score <- GET_STATE(unit_id).retention_score
  IF last_score == null: RETURN null
  last_seen  <- GET_STATE(unit_id).last_seen
  elapsed    <- at_time - last_seen
  halflife   <- GET_HALFLIFE(unit_id)
  lambda     <- ln(2) / halflife
  RETURN last_score * e^(-lambda * elapsed)

GET_HALFLIFE(unit_id) ->
  sv_tier <- structural_value_tier[unit_id]
  IF sv_tier == 1: RETURN 14 days
  IF sv_tier == 4: RETURN 28 days
  RETURN 21 days

DETECT_TRIGGER(unit_id, evidence) ->
  state <- GET_STATE(unit_id)
  log   <- evidence_log[unit_id]

  IF evidence.latency < FAST_THRESHOLD
     AND evidence.outcome == incorrect:
    RETURN CT.1

  recent_trained <- log.last(5).filter(
    format IN GET_TRAINED_FORMATS(unit_id)
    AND outcome == correct
  )
  recent_new <- log.last(3).filter(
    format NOT IN GET_TRAINED_FORMATS(unit_id)
    AND outcome == incorrect
  )
  IF recent_trained.length >= 3
     AND recent_new.length >= 2:
    RETURN CT.2

  guided_correct <- log.filter(
    demand_level == L2 AND outcome == correct
  )
  free_incorrect <- log.filter(
    demand_level == L3 AND outcome == incorrect
  )
  IF guided_correct.length >= 3
     AND free_incorrect.length >= 2:
    RETURN CT.3

  IF state.retention_score != null
     AND state.retention_score < RETENTION_THRESHOLD
     AND state.mastery_level >= MASTERY_THRESHOLD:
    RETURN CT.4

  IF unit_in_fluency_scope(unit_id)
     AND evidence.conceptual_correct == TRUE
     AND evidence.fluent == FALSE:
    RETURN CT.5

  RETURN null

UNIT_CONCLUDED(unit_id) ->
  state <- GET_STATE(unit_id)

  // criterion 1: mastery evidence at adequate level
  IF NOT MASTERY_CONCLUDED(unit_id):
    RETURN FAIL(reason=INSUFFICIENT_MASTERY,
      current=state.mastery_level,
      required=MASTERY_THRESHOLD)

  // criterion 2: retention after interval
  IF NOT RETENTION_VERIFIED(unit_id):
    RETURN FAIL(reason=RETENTION_NOT_VERIFIED,
      note="requer avaliacao apos intervalo minimo")

  // criterion 3: transfer if in transfer scope
  IF unit_in_transfer_scope(unit_id)
     AND NOT state.transfer_demonstrated:
    RETURN FAIL(reason=TRANSFER_NOT_DEMONSTRATED)

  // criterion 4: integration if in integration scope
  IF unit_in_integration_scope(unit_id)
     AND NOT state.integration_demonstrated:
    RETURN FAIL(reason=INTEGRATION_NOT_DEMONSTRATED)

  // criterion 5: base stable
  IF NOT BASE_STABLE(unit_id):
    RETURN FAIL(reason=UNSTABLE_BASE,
      failing=GET_FAILING_DEPS(unit_id))

  RETURN PASS

RETENTION_VERIFIED(unit_id) ->
  log <- evidence_log[unit_id]
  reviews_after_interval <- log.filter(
    type == review
    AND timestamp > first_mastery_timestamp(unit_id)
                    + MIN_RETENTION_INTERVAL
    AND outcome == correct
  )
  RETURN reviews_after_interval.length >= 1

MIN_RETENTION_INTERVAL: 3 days

  RETURN unit_id IN {KH.*}

BLOCK_CONCLUDED(block_id) ->
  units <- BLOCKS[block_id]

  // criterion 1: all units concluded
  failing_units <- units.filter(
    UNIT_CONCLUDED(u) == FAIL
  )
  IF failing_units.not_empty:
    RETURN FAIL(reason=UNITS_NOT_CONCLUDED,
      failing=failing_units)

  // criterion 2: no structural fragility
  fragile <- units.filter(
    GET_STATE(u).retention_score < RETENTION_THRESHOLD
  )
  IF fragile.not_empty:
    RETURN FAIL(reason=STRUCTURAL_FRAGILITY,
      units=fragile)

  // criterion 3: superior blocks require base blocks
  required_base <- GET_REQUIRED_BASE(block_id)
  FOR each base IN required_base:
    IF BLOCK_CONCLUDED(base) == FAIL:
      RETURN FAIL(reason=BASE_BLOCK_INSUFFICIENT,
        block=base)

  RETURN PASS

  BASE_REQUIREMENTS = {
    BLK.C: [BLK.A, BLK.B],

    BLK.E: [BLK.A, BLK.B],

    BLK.G: [BLK.C, BLK.D, BLK.E],
    BLK.H: [BLK.E, BLK.F, BLK.G]
  }
  RETURN BASE_REQUIREMENTS[block_id] OR []

CURRICULUM_CONCLUDED(student_record) ->

  // criterion 1: all blocks concluded
  failing_blocks <- BLOCKS.keys().filter(
    BLOCK_CONCLUDED(b) == FAIL
  )
  IF failing_blocks.not_empty:
    RETURN FAIL(reason=BLOCKS_NOT_CONCLUDED,
      failing=failing_blocks)

  // criterion 2: central content retention
  central <- {BLK.A, BLK.B, BLK.C, BLK.D, BLK.E}
  retention_failing <- central.filter(
    AVG_RETENTION(b) < RETENTION_THRESHOLD
  )
  IF retention_failing.not_empty:
    RETURN FAIL(reason=CENTRAL_RETENTION_INSUFFICIENT,
      blocks=retention_failing)

  // criterion 3: transfer where objective
  transfer_scope_units <- unit_list.filter(
    unit_in_transfer_scope(u)
  )
  transfer_failing <- transfer_scope_units.filter(
    NOT GET_STATE(u).transfer_demonstrated
  )
  IF transfer_failing.not_empty:
    RETURN FAIL(reason=TRANSFER_NOT_DEMONSTRATED,
      units=transfer_failing)

  // criterion 4: integration where objective
  integration_scope_units <- unit_list.filter(
    unit_in_integration_scope(u)
  )
  integration_failing <- integration_scope_units.filter(
    NOT GET_STATE(u).integration_demonstrated
  )
  IF integration_failing.not_empty:
    RETURN FAIL(reason=INTEGRATION_NOT_DEMONSTRATED,
      units=integration_failing)

  // criterion 5: terminal performance compatible
  IF NOT TERMINAL_PERFORMANCE_MET(student_record):
    RETURN FAIL(reason=TERMINAL_PERFORMANCE_NOT_MET)

  RETURN PASS

AVG_RETENTION(block_id) ->
  units  <- BLOCKS[block_id]
  scores <- units.map(
    GET_STATE(u).retention_score
  ).filter(NOT null)
  IF scores.empty: RETURN 0
  RETURN AVG(scores)
